def bresenham(x0,y0,x1,y1):
    """it will give coordinates of pixels for drawing line from (x0,y0) to (x1,y1)"""
    dx=x1-x0
    dy=y1-y0
    flag=0
    if (dy>dx):
        c=x1
        x1=y1
        y1=c
        d=x0
        x0=y0
        y0=d
        flag=1
    p=2*dy-dx
    x=x0
    y=y0
    i=0
    if flag==1:
        print(y,x)
    else:
        print(x,y)
    while i<dx:
        if p<0:
            x+=1
            p+=2*dy
        else:
            x+=1
            y+=1
            p+=2*dy-2*dx
        print(x,y) if flag==0 else print(y,x)
        i+=1
