#include<iostream>
#include <cmath>
using std::cout;
using std::endl;
int main()
{
	cout<<sqrt(23)<<endl;
	cout<<sqrt(-25)<<endl;
	cout<<pow(9,999)<<endl;
	cout<< INFINITY<<endl; // we can use constants to print these
	cout<<-INFINITY<<endl;
	cout<<NAN<<endl;

	//airthmetic operators and func
	cout<<remainder(10,3.25)<<endl;
	cout<<10%3<<endl;
	cout<<10%3.25<<endl; // will not give .25 (will generate error)

	cout<<fmod(10,3.25)<<endl;// will also work.
	cout<<fmax(10,3)<<endl; //provides highest;
	cout<<fmin(10,3)<<endl; //provide minimum of passed arguments.

	cout<<ceil(3.35); //will give us 4
	cout<<floor(3.23); //prints 3
	cout<<trunc(1.5); //will give us 1
	cout<<trunc(-1.5); // will give us 1
	cout<<floor(-1.5); // will print -2 (give us downwards value)
	cout<<round(-1.49); // will give us -1

	// follow cppreference online.
