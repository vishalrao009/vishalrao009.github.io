/******************************DO while loop******************************************
 *It is used when we want to do something at least once. like running a loop untill some condition
 is met. 
 */
#include<iostream>
using std::cout;
using std::cin;
using std::endl;
int main()
{
	std::string passwd="Password123";
	std::string guess;
	do
	{
		cin>>guess;
		cout<<"Enter the guess again"<<endl;
	}while(guess!=passwd);
}
